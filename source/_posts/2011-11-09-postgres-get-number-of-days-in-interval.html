--- layout: post title: "Postgres: get the number of days in an interval" date: 2011-11-09 comments: false categories: - postgresql ---
-- seconds / 60 = minutes / 60 = hours / 24 = days
select (((EXTRACT(EPOCH FROM INTERVAL '2 years') / 60) / 60) / 24)::integer as number_of_days
